```python
import torch

# 1.matrix dot
def matdot():
    a = torch.arange(6)
    b = torch.arange(6)
    print('a =', a)
    print('b =', b)
    print('a·b =', torch.dot(a, b))

# 2.Matrix multiplication
def matmultip():
    a = torch.rand(3, 4)
    b = torch.rand(4, 7)
    y = torch.mm(a, b)
    print('a =', a)
    print('b =', b)
    print('a*b =', y)

# 3.matrix inverse
def invmat():
    a = torch.rand(3, 3)
    b = torch.inverse(a)
    print('a =', a)
    print('a\'s inverse matrix is:\n', b)

if __name__ == '__main__':
    print('1.matrix dot')
    matdot()
    print('2.Matrix multiplication')
    matmultip()
    print('3.matrix inverse')
    invmat()
```
